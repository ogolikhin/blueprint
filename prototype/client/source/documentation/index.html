C) Page Structure - top

Page structure template_content/template/app.html is the file that include the header, sidebar, right sidebar, top menu and content container.

<!-- begin #header -->
<div ng-include src="'template/header.html'"></div>
<!-- end #header -->

<div ng-if="setting.layout.pageTopMenu">
    <!-- begin #sidebar -->
    <div ng-include src="'template/top-menu.html'"></div>
    <!-- end #sidebar -->
</div>

<div ng-if="setting.layout.pageWithoutSidebar === false">
    <!-- begin #sidebar -->
    <div ng-include src="'template/sidebar.html'"></div>
    <!-- end #sidebar -->
</div>

<!-- begin #content -->
<div id="content" class="content" ng-class="{ 
     'p-0': setting.layout.pageContentWithoutPadding, 
     'height-full': setting.layout.pageContentFullHeight
}">
    <div ui-view></div>
</div>
<!-- end #content -->

<!-- begin #sidebar-right -->
<div ng-include src="'template/sidebar-right.html'"></div>
<!-- end #sidebar-right -->

<!-- begin #theme-panel -->
<div ng-include src="'template/theme-panel.html'"></div>
<!-- end #theme-panel -->

D) Page Theme Options - top

Below is the sample of how to set the theme color option classes on sidebar, sidebar-bg and header. 

<!-- begin #sidebar -->
<div class="sidebar sidebar-primary">
  <!-- sidebar here -->
</div>
<div class="sidebar-bg sidebar-primary"></div>
<!-- end #sidebar -->

<!-- begin #header -->
<div id="header" class="header navbar navbar-primary navbar-fixed-top">
  <!-- header here -->
</div>
<!-- end #header -->

E) Page Options - top

Here is the general structure for the each case of page options.

Page Boxed Layout

<body class="boxed-layout">

or you may also set in your angular js controller

$rootScope.setting.layout.pageBoxedLayout = true;

Page Without Sidebar

<!-- begin #page-container -->
<div id="page-container" class="fade page-without-sidebar">

or you may also set in your angular js controller

$rootScope.setting.layout.pageWithoutSidebar = true;

Page With Two Sidebar

<!-- begin #page-container -->
<div id="page-container" class="fade page-with-two-sidebar">
  <!-- begin #sidebar -->
  <div id="sidebar" class="sidebar"></div>
  <div class="sidebar-bg"></div>
  <!-- end #sidebar -->
  
  <!-- begin #sidebar-right -->
  <div id="sidebar-right" class="sidebar sidebar-right"></div>
  <div class="sidebar-bg sidebar-right"></div>
  <!-- end #sidebar-right -->

or you may also set in your angular js controller

$rootScope.setting.layout.pageTwoSidebar = true;

Page With Fixed Footer

<!-- begin #page-container -->
<div id="page-container" class="fade page-with-footer page-footer-fixed">
  <!-- begin #footer -->
  <div id="footer" class="footer">
    <!-- Footer Here -->
  </div>
  <!-- end #footer -->

or you may also set in your angular js controller

$rootScope.setting.layout.pageFixedFooter = true;

Sidebar Minified

<!-- begin #page-container -->
<div id="page-container" class="fade page-sidebar-minified">
  <!-- begin #sidebar -->
  <div id="sidebar" class="sidebar">
    <!-- Sidebar Here -->
  </div>
  <div class="sidebar-bg"></div>
  <!-- end #sidebar -->

or you may also set in your angular js controller

$rootScope.setting.layout.pageSidebarMinified = true;

Sidebar Position Right

<!-- begin #page-container -->
<div id="page-container" class="fade page-with-right-sidebar">
  <!-- begin #sidebar -->
  <div id="sidebar" class="sidebar">

or you may also set in your angular js controller

$rootScope.setting.layout.pageRightSidebar = true;

Page with Top Menu

<!-- begin #page-container -->
<div id="page-container" class="fade page-with-top-menu">
  <!-- begin #sidebar -->
  <div id="sidebar" class="sidebar">

or you may also set in your angular js controller

$rootScope.setting.layout.pageTopMenu = true;

Page Loading HTML

If you wish to remove the page loading screen, you can remove the page loader and the fade class in page container.

Loading 
<!-- begin #page-loader -->
<div id="page-loader" class="fade in"><span class="spinner"></span></div>
<!-- end #page-loader -->

<!-- begin #page-container -->
<div id="page-container" class="fade">
</div>
<!-- end #page-container -->
After Loading 
<!-- begin #page-loader -->
<div id="page-loader" class="fade"><span class="spinner"></span></div>
<!-- end #page-loader -->

<!-- begin #page-container -->
<div id="page-container" class="fade in">
</div>
<!-- end #page-container -->
